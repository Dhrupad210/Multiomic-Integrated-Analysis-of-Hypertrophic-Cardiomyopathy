# SECTION A: UTILITY FUNCTIONS
# ─────────────────────────────────────────────────────────────

FUNCTION safe_morans_i(coords, values, max_n=600):
    IF len(coords) > max_n:
        RANDOMLY subsample to max_n spots
    COMPUTE pairwise distance matrix dm
    SET cutoff = 10th percentile of dm[dm > 0]
    BUILD weights matrix w = 1/dm where 0 < dm ≤ cutoff, else 0
    ROW-STANDARDISE w
    RETURN Moran's I = (n / sum(w)) * sum(w * outer(deviations)) / sum(deviations²)

FUNCTION spatial_smoothing(coords, values, sigma=5):
    CREATE 100×100 regular grid over coord bounding box
    INTERPOLATE scattered values onto grid using cubic interpolation
    APPLY Gaussian filter with sigma
    RETURN (grid_x, grid_y, smoothed_values)

FUNCTION detect_boundaries(coords, values):
    INTERPOLATE values onto 100×100 grid (cubic)
    COMPUTE Sobel edge detection (horizontal + vertical gradients)
    RETURN gradient magnitude

FUNCTION score_module(adata, gene_list, score_name):
    FILTER gene_list to genes present in adata.var_names
    CALL scanpy.tl.score_genes(adata, present_genes, score_name)
    RETURN count of genes used

FUNCTION get_gene(expr_matrix, gene, obs_names):
    IF gene IN expr_matrix.index:
        RETURN expr_matrix.loc[gene].reindex(obs_names, fill_value=0)
    ELSE:
        RETURN zeros(len(obs_names))

FUNCTION get_genes_mean(expr_matrix, genes, obs_names):
    avail = [g for g in genes if g IN expr_matrix.index]
    IF avail is empty: RETURN (None, [])
    mat = expr_matrix.loc[avail].reindex(columns=obs_names, fill_value=0)
    RETURN (mat.mean(axis=0), avail)

FUNCTION stage_means_series(values, stage, unique_stages, valid_mask):
    FOR each stage s IN unique_stages:
        mask = (stage == s) AND valid_mask
        IF sum(mask) > 0:
            RETURN mean(values[mask])
        ELSE:
            RETURN NaN
